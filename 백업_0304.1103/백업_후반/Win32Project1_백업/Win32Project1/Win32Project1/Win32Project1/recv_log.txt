	/**
	*	App running
	*/
	function runApp(appPath, appStorePath, client){
		var checkClient;
	
		//안드로이드, 아이폰, 아이패드 확인
		if(uAgent.indexOf('iphone') > 0  || uAgent.indexOf('ipod') > 0 || uAgent.indexOf('ipad') > 0 ){
			checkClient = "i";
		}else if(uAgent.indexOf('android') > 0){
			checkClient = "a";
		}else{
			alert("iOS, 안드로이드에서 사용할 수 있습니다.");
			return;
		}
		
		if(client == checkClient){
			if(uAgent.indexOf('iphone') > 0  || uAgent.indexOf('ipod') > 0 || uAgent.indexOf('ipad') > 0 ){
				check_app.location.href = appPath;
				setTimeout("linkIphone('"+appStorePath+"')", 1000*1);
			}else if(uAgent.indexOf('android') > 0){
				check_app.location.href = appPath;
				setTimeout("linkAndroid('"+appStorePath+"')", 1000*1);
			}else{
				
				if(client == "i"){
					alert("iOS(아이폰, 아이패드)에서만 사용할 수 있습니다.");
				}else if(client=="a"){
					alert("안드로이드OS 에서만 사용할 수 있습니다.");
				}
				
			}
		}
	}
	
	function linkIphone(link){

			alert("iOS(아이폰, 아이패드)에서만 접속이 가능합니다.");
		}
	}
